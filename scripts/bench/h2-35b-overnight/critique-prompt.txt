+/**
+ * Sequential-unit planning for critiques (workstream A.3).
+ *
+ * The local servers expose ONE slot each (35b: -np 1), so "concurrent" units
+ * already serialize per REQUEST server-side — but critic units are multi-turn
+ * agents, and interleaved turns from two conversations evict each other's
+ * divergent-middle KV on the shared slot; only the byte-identical
+ * system+artifact prefix (A.1) survives an interleave. `critic.sequentialUnits`
+ * instead runs each model's units strictly serially (model groups still run
+ * concurrently), so every conversation keeps its own cache warm across its
+ * turns and hands the shared prefix to the next unit when it finishes.
+ *
+ * Within a model group the plan is BODY-major: units reviewing the same
+ * artifact/chunk body run back-to-back, because consecutive same-body units
+ * share system prompt + artifact, while same-critic-next-chunk units share
+ * only the system prompt.
+ *
+ * Pure module — no pi imports; tested by schedule.test.mjs.
+ */

+/** Model resolution mirror of subagent/run.ts: forceModel > settings
+ *  override > agent frontmatter. Returns a stable grouping key, never
+ *  undefined — unresolved units group together under "(default)". */
+export function unitModelKey(
+	agent: { model?: string; forceModel?: string } | undefined,
+	settingsOverride: string | undefined,
+): string {
+	const pick = (v: string | undefined) => (typeof v === "string" && v.trim() ? v.trim() : undefined);
+	return pick(agent?.forceModel) ?? pick(settingsOverride) ?? pick(agent?.model) ?? "(default)";
+}

+/**
+ * Partition units into per-model groups (first-seen model order, so the
+ * caller's concurrency treats groups like it treated units); within each

+export function planSequentialGroups<T extends { body: string }>(
+	units: T[],
